Step 1:
Install and update apps
sudo apt update && sudo apt upgrade -y
sudo apt install nginx python3-pip python3-venv git curl -y
sudo apt install -y nodejs
sudo npm install pm2 -g

Step 2:
Clone your repo and install running development server on background
git clone <your-backend-repo>

For Next Js
pm2 start npm --name frontend -- start
pm2 save

For Django
python3 manage.py collectstatic
nohup python3 manage.py  runserver > server.log 2>&1 &

Step 3:
Setup Nginx
sudo nano /etc/nginx/sites-available/b-soft

Then paste it
server {
    listen 80;
    server_name b-soft.xyz;

    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }
}

server {
    listen 80;
    server_name b-soft.xyz www.b-soft.xyz;

    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }
}


server {
    listen 80;
    server_name api.b-soft.xyz;  

    location = /favicon.ico { access_log off; log_not_found off; }
    location /static/ {
        root /home/ubuntu/b-soft/backend;
    }
    
    location / {
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }

    error_log /var/log/nginx/test-django-error.log;
    access_log /var/log/nginx/test-django-access.log;
}


server {
    listen 80;
    server_name dashboard.b-soft.xyz;

    location / {
        proxy_pass http://localhost:3001;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }
}

Then file permission

sudo chmod o+x /home
sudo chmod o+x /home/ubuntu
sudo chmod o+x /home/ubuntu/b-soft
sudo chmod o+x /home/ubuntu/b-soft/backend
sudo chmod -R o+rx /home/ubuntu/b-soft/backend/static

then enable Nginx
sudo ln -s /etc/nginx/sites-available/b-soft /etc/nginx/sites-enabled/
sudo nginx -t
sudo systemctl restart nginx
